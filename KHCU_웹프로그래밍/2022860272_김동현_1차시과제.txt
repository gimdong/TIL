 <!DOCTYPE html><!-- !DOCTYPE html : 이 문서는 html로 작성 되었음을 선언 합니다. -->
 <html>  <!-- html : html 문서의 시작과 끝을 나타냅니다. -->
   <head> <!-- head : 문서의 메타데이터(문서 제목, 문자 인코딩, 스타일시트, 외부 파일 링크 등)를 포함하는 영역 -->
     <meta charset="UTF-8"> <!-- meta charset="UTF-8" : 문자 인코딩을 UTF-8로 설정 -->
       <title>HTML 태그의 속성</title> <!-- title : 웹 브라우저 탭에 표시될 문서 제목 -->
   </head>
   <body> <!-- body : 실제 웹 페이지에 표시되는 내용을 포함하는 영역 -->
    <h3>웹프로그래밍 HTML 연습</h3> <!-- h3 : 제목을 나타내는 태그, 숫자가 작아질수록 제목의 수준이 높아짐 -->
    <p>HTML 문서 작성 연습</p> <!-- p : 단락을 나타내는 태그 -->
    <a href="https://www.naver.com" target="_blank"> <!-- a : 하이퍼링크를 생성하는 태그, href 속성은 링크 URL, target="_blank"는 새 탭에서 열기 -->
       네이버 사이트로 이동</a> <!-- 앵커 텍스트: 사용자가 클릭할 수 있는 부분 -->
   </body>
